import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";

## Audio controls

<Tabs>
  <TabItem value="audioOn" label="Sound On" default>
    ```jsx
    showToast({
message: "New message",
type: "success",
audio: {
enabled: true,
},
});

````

  </TabItem>
  <TabItem value="audioOff" label="Sound Off">
        ```jsx
    showToast({
  message: "New message",
  type: "success",
  audio: {
    enabled: false,
  },
});
````

  </TabItem>
  <TabItem value="soundAdjust" label="Volume adjust">
        ```jsx
    showToast({
message: "New message",
type: "success",
audio: {
volume: 0.9, // max 1
},
});

````

  </TabItem>

    <TabItem value="cooldown" label="Cooldown">
        ```jsx
    showToast({
  message: "New message",
  type: "success",
  audio: {
    cooldown: 800, // ms
  },
});
````

  </TabItem>

</Tabs>

## Default sounds

<Tabs>
  <TabItem value="success" label="On Success" default>
    ```jsx
    const { showToast, sounds } = useToast();

    showToast({
message: "New message",
type: "success",
audio: {
audioFile: sounds.success
},
});

````

  </TabItem>
