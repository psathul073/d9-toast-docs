                            // Delete logic
                            toastIdRef && removeToast(toastIdRef);
                        }}
                    >
                        Delete
                    </button>
                </div>
            </div>
        );

        const id = showToast({
            message: model,
            position: "center",
            autoClose: false,
            title: false,
        });

        toastIdRef = id;
    };
