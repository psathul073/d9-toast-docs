import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";
import { ToastButton } from "@site/src/components/ToastButton";

## Audio controls

<Tabs>
  <TabItem value="audioOn" label="Sound On" default>

```jsx
toast.success("New message", { audio: { enabled: true } });
````
Click the button below to trigger a toast:

<ToastButton
  type="success"
  message="New message"
/>

  </TabItem>
  <TabItem value="audioOff" label="Sound Off">

```jsx
toast.success("New message", { audio: { enabled: false } });
````

Click the button below to trigger a toast:

<ToastButton
  type="success"
  message="New message"
  audio={{ enabled: false}}
/>
  </TabItem>

  <TabItem value="soundAdjust" label="Volume adjust">
        
```jsx
toast.success("New message", { audio: { volume: 0.9, // max 1 } });
````
Click the button below to trigger a toast:

<ToastButton
  type="success"
  message="New message"
  audio={{ volume: 0.9,}}
/>
  </TabItem>

<TabItem value="cooldown" label="Cooldown">

```jsx
toast.success("New message", { audio: { cooldown: 800, // ms } });
````

Click the button below to trigger a toast:

<ToastButton type="success" message="New message" audio={{ cooldown: 800 }} />

  </TabItem>

</Tabs>
